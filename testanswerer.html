Test answerer

Has it ever happened to you that you didn't know the questions for the test?
I have a solution for this situation!

What you will need:

  • arduino
  • some wires
  • a button
  • 3 LEDs
  • 4 resistor
  • some stickers or little papers

Circuit:

We connect the button to D5 and the LEDs to D2, D3 and D4.

Code:


          const int button = 5;  // Button to D5
          const int ledR = 2;    // First LED to D2
          const int ledG = 3;    // Second LED to D3
          const int ledY = 4;    // Third LED to D4
          
          int buttonState;
          int answer;
          
          void setup() {
            pinMode(button, INPUT);
            pinMode(ledR, OUTPUT);
            pinMode(ledG, OUTPUT);
            pinMode(ledY, OUTPUT);
          }
          
          void loop(){
            buttonState = digitalRead(button); // We save the value of the button in this variable. It can be only 0 or 1.
            
            if (buttonState == 1){
              for (int i = 1; i <= 3; i++){
                digitalWrite(ledR, 1);
                digitalWrite(ledG, 0);
                digitalWrite(ledY, 0);
                delay(100);
                digitalWrite(ledR, 0);
                digitalWrite(ledG, 1);
                digitalWrite(ledY, 0);
                delay(100);
                digitalWrite(ledR, 0);
                digitalWrite(ledG, 0);
                digitalWrite(ledY, 1);
                delay(100);
              } // We make the LEDs flash.
          
              answer = random(1, 3+1); // We generate a random number from 1 to 3.
          
              if (answer == 1){
                digitalWrite(ledR, 1);
                digitalWrite(ledG, 0);
                digitalWrite(ledY, 0);
              }
              else if (answer == 2){
                digitalWrite(ledR, 0);
                digitalWrite(ledG, 1);
                digitalWrite(ledY, 0);
              }
              else{
                digitalWrite(ledR, 0);
                digitalWrite(ledG, 0);
                digitalWrite(ledY, 1);
              }// The illuminated LED shows the number of the answer.
            }
          }
