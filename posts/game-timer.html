  • your friend's computer

  • a button
  • a passive buzzer

Component Arduino pin
Button D2
Buzzer D4
Joystick (URX) A0
Joystick (URY) A1

Oled (SCL) SCL or A5
Oled (SDA) SDA or A4

    #include <Wire.h>
    #include <Adafruit_SSD1306.h>
    #include <Adafruit_GFX.h>

    #define SCREEN_I2C_ADDRESS 0x3C // or try 0x3D
    #define SCREEN_WIDTH 128     
    #define SCREEN_HEIGHT 64     
    #define RESET_PIN -1 

    Adafruit_SSD1306 display(SCREEN_WIDTH, SCREEN_HEIGHT, &Wire, RESET_PIN);



    int urx = A0;
    int ury = A1;
    int joystickbtn = 3;
    int button = 2;
    int buzzer = 4;

    int readUrx;
    int readUry;
    int joystickbtnState;
    int buttonState;

    int minute = 30;
    int hour = 1; 
    int second = 0;
    int chooseUnit = 1;
    bool menuMode = 1;
    bool countdown = 0;



    void setup() {
    // We set INPUT_PULLUP instead of INPUT because that way we won't need a pull-up or pull-down resistor.
    pinMode(button, INPUT_PULLUP);
    pinMode(joystickbtn, INPUT_PULLUP);
    pinMode(urx, INPUT);
    pinMode(ury, INPUT);
    pinMode(buzzer, OUTPUT);

    display.begin(SSD1306_SWITCHCAPVCC,SCREEN_I2C_ADDRESS);
    display.setTextColor(WHITE);
    display.setTextSize(2);

    Serial.begin(9600);
    }



    void loop() {
    // If we don't begin the countdown(= we don't press the button) "countdown" will be 0, so we can chage the hours, minutes and seconds.
    if (countdown == 0){
        readUrx = analogRead(urx);
        readUry = analogRead(ury);
        buttonState = digitalRead(button);
        joystickbtnState = digitalRead(joystickbtn);

        // If "menuMode" is 1, we can choose from the units (hours, minutes, seconds)
        // We can change the value of "menuMode" by pressing the button of the joystick.
        if (menuMode == 1){ 
        if (readUrx > 700){
            chooseUnit ++;
        }
        else if (readUrx < 350){
            chooseUnit --;
        }

        if (chooseUnit > 3){
            chooseUnit = 1;
        }
        if (chooseUnit < 1){
            chooseUnit = 3;
        }
        }

        //After pressing the switch of the joystick we can increase or decrease the hours, minutes or seconds
        else{
        if (readUry > 700){
            if (chooseUnit == 1 and hour > 0){
            hour --;
            }
            else if (chooseUnit == 2 and minute > 0){
            minute --;
            }
            else if (chooseUnit == 3 and second > 0){
            second--;
            }
        }
        else if (readUry < 350){
            if (chooseUnit == 1 and hour < 5){
            hour ++;
            }
            else if (chooseUnit == 2 and minute < 59){
            minute ++;
            }
            else if (chooseUnit == 3 and minute < 59){
            second ++;
            }
        }
        }

        if (joystickbtnState == 0){
        tone(buzzer, 900, 50);
        menuMode =! menuMode;
        }

        // We can start the countdown by pressing the button
        if (buttonState == 0){
        if(hour == 0 and minute == 0 and second == 0){
            tone(buzzer, 700, 50);
            delay(100);
            tone(buzzer, 300, 100);
            countdown = 0;
        }
        else{
            tone(buzzer, 700, 50);
            delay(100);
            tone(buzzer, 1000, 100);
            countdown = 1;
        }
        }

        delay(100);
    }
    else{
        second--;
        if (second == -1){
        second = 59;
        minute --;
        }
        if (minute < 0 and hour > 0){
        hour --;
        minute = 59;
        }
        if (second == 59 and minute == -1 and hour == 0){
        display.clearDisplay();
        display.setCursor(0, 0);
        display.print("TIME IS UP");
        display.display();
        delay(1000);
        Serial.println(1); // If the time is up the arduino sends a value to the computer.
        minute = 30;
        hour = 1; 
        second = 0;
        countdown = 0;
        }
        delay(1000);
    }
    
    //We display the hours, minutes, seconds and the rectangle on the oled display.
    display.clearDisplay();
    display.setCursor(29, 0);
    display.print("h  m  s");
    if (hour > 9){
        display.setCursor(17, 25);
    }
    else{
        display.setCursor(29,25);
    }
    display.print(hour);
    display.print(":");
    if (minute > 9){
        display.print(minute);
    }
    else{
        display.print(' ');
        display.print(minute);
    }
    display.print(":");
    if (second > 9){
        display.print(second);
    }
    else{
        display.print(' ');
        display.print(second);
    }
    
    if (countdown == 0){
        if (chooseUnit == 1){
        display.drawBitmap(0, 0, recthour, SCREEN_WIDTH, SCREEN_HEIGHT, 1);
        }
        else if (chooseUnit == 2){
        display.drawBitmap(0, 0, rectmin, SCREEN_WIDTH, SCREEN_HEIGHT, 1);
        }
        else if (chooseUnit == 3){
        display.drawBitmap(0, 0, rectsec, SCREEN_WIDTH, SCREEN_HEIGHT, 1);
        }
    }
    display.display();

    }
